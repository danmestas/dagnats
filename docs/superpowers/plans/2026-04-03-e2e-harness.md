# E2E Test Harness Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the test harness infrastructure (`e2e/harness/`) that enables writing E2E tests once and running them against embedded, local cluster, and supercluster NATS topologies.

**Architecture:** Three topology providers implement a common interface. `RunE2E` iterates over enabled topologies (controlled by `E2E_TOPOLOGY` env var) and runs each test against each one. Shared helpers handle the boilerplate of starting orchestrators, workers, and polling for run completion. The supercluster provider spins up 5 in-process NATS servers with cluster routes, gateways, and a leaf node.

**Tech Stack:** Go, `nats-server/v2` embedded Go library, `nats.go` client

**Spec:** `docs/superpowers/specs/2026-04-03-e2e-test-suite-design.md`

// e2e/harness/topology.go
// Defines the Topology interface for E2E test infrastructure providers.
// Each provider manages NATS server lifecycle and provides connections.
package harness

import (
	"testing"

	"github.com/nats-io/nats.go"
)

// Topology provides a NATS connection for E2E tests. Simple topologies
// (embedded, local cluster) only need Connect. The supercluster
// topology also supports infrastructure manipulation for resilience tests.
type Topology interface {
	// Name returns the topology name for test output (e.g., "embedded").
	Name() string

	// Connect returns a NATS connection to this topology. Registers
	// cleanup via t.Cleanup. Panics on connection failure — a broken
	// topology is a test infrastructure error, not a test result.
	Connect(t *testing.T) *nats.Conn

	// Setup creates NATS streams and KV buckets required by DagNats.
	// Called once per test suite. Panics on failure.
	Setup(t *testing.T, nc *nats.Conn)
}

// Resilient extends Topology with infrastructure manipulation methods
// for resilience testing. Only the supercluster topology implements this.
type Resilient interface {
	Topology

	// KillNode shuts down a named node. Valid names: "a1", "a2", "b1", "b2".
	// Panics on unknown node name (programmer error).
	KillNode(name string) error

	// RestartNode restarts a previously killed node. Waits up to 10s
	// for cluster rejoin. Panics on unknown name or rejoin timeout.
	RestartNode(name string) error

	// DisconnectLeaf shuts down the leaf node.
	DisconnectLeaf() error

	// ReconnectLeaf restarts the leaf node. Waits up to 10s for
	// leaf connection to establish. Panics on timeout.
	ReconnectLeaf() error
}
```

// e2e/harness/harness.go
// RunE2E runs a test function against all enabled topologies.
// Topology selection via E2E_TOPOLOGY env var: "embedded",
// "local_cluster", "supercluster", or empty for all.
package harness

import (
	"os"
	"testing"

	"github.com/nats-io/nats.go"
)

// E2ETest is the signature for topology-agnostic E2E tests.
type E2ETest func(t *testing.T, nc *nats.Conn)

// RunE2E runs the test function against each enabled topology.
// Panics if no topologies are enabled (e.g., typo in E2E_TOPOLOGY).
func RunE2E(t *testing.T, test E2ETest) {
	t.Helper()
	topos := enabledTopologies(t)
	if len(topos) == 0 {
		t.Fatal(
			"RunE2E: no topologies enabled — check E2E_TOPOLOGY",
		)
	}
	for _, topo := range topos {
		topo := topo
		t.Run(topo.Name(), func(t *testing.T) {
			nc := topo.Connect(t)
			topo.Setup(t, nc)
			test(t, nc)
		})
	}
}

// enabledTopologies returns the topologies selected by E2E_TOPOLOGY.
// Empty string means all topologies. Panics on unrecognized values.
func enabledTopologies(t *testing.T) []Topology {
	t.Helper()
	env := os.Getenv("E2E_TOPOLOGY")

	all := []Topology{
		NewEmbedded(),
		NewLocalCluster(),
		NewSupercluster(),
	}

	if env == "" {
		return all
	}

	valid := map[string]bool{
		"embedded":      true,
		"local_cluster": true,
		"supercluster":  true,
	}
	if !valid[env] {
		t.Fatalf(
			"enabledTopologies: unknown E2E_TOPOLOGY=%q "+
				"(valid: embedded, local_cluster, supercluster)",
			env,
		)
	}

	for _, topo := range all {
		if topo.Name() == env {
			return []Topology{topo}
		}
	}
	return nil
}
```

// e2e/harness/helpers.go
// Shared test helpers for E2E tests. Eliminates boilerplate around
// starting orchestrators, registering workflows, polling for status,
// and asserting event history.
package harness

import (
	"context"
	"fmt"
	"sync/atomic"
	"testing"
	"time"

	"github.com/danmestas/dagnats/api"
	"github.com/danmestas/dagnats/dag"
	"github.com/danmestas/dagnats/observe"
	"github.com/danmestas/dagnats/protocol"
	"github.com/danmestas/dagnats/worker"
	"github.com/nats-io/nats.go"
)

var nameCounter atomic.Int64

// UniqueName returns a test-unique name to prevent key collisions
// when tests share KV buckets across topologies.
func UniqueName(t *testing.T, base string) string {
	t.Helper()
	if base == "" {
		panic("UniqueName: base must not be empty")
	}
	n := nameCounter.Add(1)
	return fmt.Sprintf("%s-%d", base, n)
}

// NewTestService creates an api.Service with noop telemetry.
// Does NOT call SetupAll — the topology's Setup method handles that.
func NewTestService(
	t *testing.T, nc *nats.Conn,
) *api.Service {
	t.Helper()
	if nc == nil {
		panic("NewTestService: nc must not be nil")
	}
	tel := observe.NewNoopTelemetry()
	return api.NewService(nc, tel)
}

// RegisterAndStart registers a workflow and starts a run.
// Returns the run ID. Fails the test on error.
func RegisterAndStart(
	t *testing.T, svc *api.Service,
	wfDef dag.WorkflowDef, input []byte,
) string {
	t.Helper()
	if svc == nil {
		panic("RegisterAndStart: svc must not be nil")
	}
	if wfDef.Name == "" {
		panic("RegisterAndStart: workflow name must not be empty")
	}
	ctx := context.Background()
	if err := svc.RegisterWorkflow(ctx, wfDef); err != nil {
		t.Fatalf("RegisterWorkflow %q: %v", wfDef.Name, err)
	}
	runID, err := svc.StartRun(ctx, wfDef.Name, input)
	if err != nil {
		t.Fatalf("StartRun %q: %v", wfDef.Name, err)
	}
	return runID
}

// WaitForRunStatus polls until the run reaches the expected status.
// Uses 250ms poll interval (generous for cross-cluster propagation).
// Fails the test on timeout.
func WaitForRunStatus(
	t *testing.T, svc *api.Service,
	runID string, status dag.RunStatus, timeout time.Duration,
) dag.WorkflowRun {
	t.Helper()
	if svc == nil {
		panic("WaitForRunStatus: svc must not be nil")
	}
	if runID == "" {
		panic("WaitForRunStatus: runID must not be empty")
	}
	ctx := context.Background()
	deadline := time.After(timeout)
	for {
		run, err := svc.GetRun(ctx, runID)
		if err == nil && run.Status == status {
			return run
		}
		select {
		case <-deadline:
			lastStatus := "unknown"
			if err == nil {
				lastStatus = string(run.Status)
			}
			t.Fatalf(
				"WaitForRunStatus: run %q did not reach %q "+
					"within %s (last: %s)",
				runID, status, timeout, lastStatus,
			)
		case <-time.After(250 * time.Millisecond):
		}
	}
}

// SubscribeWorker creates a worker, registers the handler, starts it,
// and registers cleanup. Returns the worker for chaining.
func SubscribeWorker(
	t *testing.T, nc *nats.Conn,
	taskName string, handler worker.HandlerFunc,
) *worker.Worker {
	t.Helper()
	if nc == nil {
		panic("SubscribeWorker: nc must not be nil")
	}
	if taskName == "" {
		panic("SubscribeWorker: taskName must not be empty")
	}
	tel := observe.NewNoopTelemetry()
	w := worker.NewWorker(nc, tel)
	w.Handle(taskName, handler)
	w.Start()
	t.Cleanup(func() { w.Stop() })
	return w
}

// AssertHistoryContains verifies the run's history contains the
// expected event types in order. Events may have other events
// between them — this checks subsequence, not exact match.
func AssertHistoryContains(
	t *testing.T, svc *api.Service,
	runID string, expected ...protocol.EventType,
) {
	t.Helper()
	if svc == nil {
		panic("AssertHistoryContains: svc must not be nil")
	}
	if runID == "" {
		panic("AssertHistoryContains: runID must not be empty")
	}
	if len(expected) == 0 {
		panic("AssertHistoryContains: expected must not be empty")
	}
	ctx := context.Background()
	events, err := svc.ListRunEvents(ctx, runID, false)
	if err != nil {
		t.Fatalf("ListRunEvents %q: %v", runID, err)
	}

	ei := 0
	for _, evt := range events {
		if ei < len(expected) &&
			protocol.EventType(evt.Type) == expected[ei] {
			ei++
		}
	}
	if ei < len(expected) {
		t.Fatalf(
			"AssertHistoryContains: run %q missing event %q "+
				"(matched %d/%d expected events)",
			runID, expected[ei], ei, len(expected),
		)
	}
}

	"testing"

	"github.com/nats-io/nats.go"

	return &LocalClusterTopology{}

	return "local_cluster"

// Connect starts a NATS server with production-like config and
// returns a connection. Cleaned up via t.Cleanup.
func (l *LocalClusterTopology) Connect(
	t *testing.T,
) *nats.Conn {
	t.Helper()
	opts := &natsserver.Options{
		Host:               "127.0.0.1",
		Port:               -1,
		JetStream:          true,
		StoreDir:           t.TempDir(),
		JetStreamMaxMemory: 256 * 1024 * 1024, // 256MB
		JetStreamMaxStore:  512 * 1024 * 1024,  // 512MB
		MaxPayload:         8 * 1024 * 1024,    // 8MB
	}
	ns, err := natsserver.NewServer(opts)
	if err != nil {
		t.Fatalf("local_cluster: create server: %v", err)
	}
	ns.Start()
	if !ns.ReadyForConnections(5 * time.Second) {
		t.Fatal("local_cluster: server not ready after 5s")
	}
	t.Cleanup(func() {
		ns.Shutdown()
		ns.WaitForShutdown()
	})

	nc, err := nats.Connect(ns.ClientURL())
	if err != nil {
		t.Fatalf("local_cluster: connect: %v", err)
	}
	t.Cleanup(func() { nc.Close() })
	return nc
}

	t *testing.T, nc *nats.Conn,
) {
	t.Helper()
	if err := natsutil.SetupAll(nc); err != nil {
		t.Fatalf("local_cluster Setup: %v", err)
	}
}

// Supercluster topology provider. Starts 5 in-process NATS servers:
// Cluster A (2 nodes) + Cluster B (2 nodes) + 1 leaf node connecting
// to Cluster A. Gateways link the two clusters. Tests connect through
// the leaf to exercise the full routing path.
package harness

import (
	"fmt"
	"net"
	"net/url"
	"testing"
	"time"

	"github.com/danmestas/dagnats/natsutil"
	natsserver "github.com/nats-io/nats-server/v2/server"
	"github.com/nats-io/nats.go"
)

// SuperclusterTopology manages a 5-node NATS supercluster.
type SuperclusterTopology struct {
	clusterA [2]*natsserver.Server
	clusterB [2]*natsserver.Server
	leaf     *natsserver.Server
	opts     map[string]*natsserver.Options
}

// NewSupercluster creates a supercluster topology provider.
func NewSupercluster() *SuperclusterTopology {
	return &SuperclusterTopology{
		opts: make(map[string]*natsserver.Options),
	}
}

func (s *SuperclusterTopology) Name() string {
	return "supercluster"
}

// Connect starts all 5 servers and returns a connection to the leaf.
func (s *SuperclusterTopology) Connect(
	t *testing.T,
) *nats.Conn {
	t.Helper()
	s.startCluster(t)

	nc, err := nats.Connect(
		s.leaf.ClientURL(),
		nats.MaxReconnects(10),
		nats.ReconnectWait(500*time.Millisecond),
	)
	if err != nil {
		t.Fatalf("supercluster: connect to leaf: %v", err)
	}
	t.Cleanup(func() { nc.Close() })
	return nc
}

// Setup creates the NATS streams and KV buckets DagNats requires.
func (s *SuperclusterTopology) Setup(
	t *testing.T, nc *nats.Conn,
) {
	t.Helper()
	if err := natsutil.SetupAll(nc); err != nil {
		t.Fatalf("supercluster Setup: %v", err)
	}
}

func (s *SuperclusterTopology) startCluster(t *testing.T) {
	t.Helper()

	// Allocate all ports upfront.
	// 5 client + 4 cluster + 4 gateway + 2 leaf listener = 15.
	ports := allocatePorts(t, 15)

	// Cluster A: 2 nodes.
	aClientPorts := [2]int{ports[0], ports[1]}
	aClusterPorts := [2]int{ports[2], ports[3]}
	aGatewayPorts := [2]int{ports[4], ports[5]}
	aLeafPorts := [2]int{ports[6], ports[7]}

	// Cluster B: 2 nodes.
	bClientPorts := [2]int{ports[8], ports[9]}
	bClusterPorts := [2]int{ports[10], ports[11]}
	bGatewayPorts := [2]int{ports[12], ports[13]}

	// Leaf node.
	leafClientPort := ports[14]

	// Build cluster route URLs.
	aRoutes := []*url.URL{
		parseURL(t, aClusterPorts[0]),
		parseURL(t, aClusterPorts[1]),
	}
	bRoutes := []*url.URL{
		parseURL(t, bClusterPorts[0]),
		parseURL(t, bClusterPorts[1]),
	}

	// Gateway configs — each cluster knows about the other.
	aGateways := []*natsserver.RemoteGatewayOpts{{
		Name: "cluster-b",
		URLs: []*url.URL{
			parseURL(t, bGatewayPorts[0]),
			parseURL(t, bGatewayPorts[1]),
		},
	}}
	bGateways := []*natsserver.RemoteGatewayOpts{{
		Name: "cluster-a",
		URLs: []*url.URL{
			parseURL(t, aGatewayPorts[0]),
			parseURL(t, aGatewayPorts[1]),
		},
	}}

	// Start Cluster A nodes (with leaf node listener for leaf connection).
	for i := 0; i < 2; i++ {
		name := fmt.Sprintf("a%d", i+1)
		opts := &natsserver.Options{
			Host:               "127.0.0.1",
			Port:               aClientPorts[i],
			ServerName:         name,
			JetStream:          true,
			StoreDir:           t.TempDir(),
			JetStreamMaxMemory: 128 * 1024 * 1024,
			JetStreamMaxStore:  256 * 1024 * 1024,
			Cluster: natsserver.ClusterOpts{
				Name: "cluster-a",
				Host: "127.0.0.1",
				Port: aClusterPorts[i],
			},
			Routes: aRoutes,
			Gateway: natsserver.GatewayOpts{
				Name: "cluster-a",
				Host: "127.0.0.1",
				Port: aGatewayPorts[i],
				Gateways: aGateways,
			},
			LeafNode: natsserver.LeafNodeOpts{
				Host: "127.0.0.1",
				Port: aLeafPorts[i],
			},
		}
		s.opts[name] = opts
		s.clusterA[i] = startServer(t, opts)
	}

	// Start Cluster B nodes.
	for i := 0; i < 2; i++ {
		name := fmt.Sprintf("b%d", i+1)
		opts := &natsserver.Options{
			Host:               "127.0.0.1",
			Port:               bClientPorts[i],
			ServerName:         name,
			JetStream:          true,
			StoreDir:           t.TempDir(),
			JetStreamMaxMemory: 128 * 1024 * 1024,
			JetStreamMaxStore:  256 * 1024 * 1024,
			Cluster: natsserver.ClusterOpts{
				Name: "cluster-b",
				Host: "127.0.0.1",
				Port: bClusterPorts[i],
			},
			Routes: bRoutes,
			Gateway: natsserver.GatewayOpts{
				Name: "cluster-b",
				Host: "127.0.0.1",
				Port: bGatewayPorts[i],
				Gateways: bGateways,
			},
		}
		s.opts[name] = opts
		s.clusterB[i] = startServer(t, opts)
	}

	// Start leaf node connecting to Cluster A's leaf listener port.
	leafOpts := &natsserver.Options{
		Host:       "127.0.0.1",
		Port:       leafClientPort,
		ServerName: "leaf",
		LeafNode: natsserver.LeafNodeOpts{
			Remotes: []*natsserver.RemoteLeafOpts{{
				URLs: []*url.URL{
					parseURL(t, aLeafPorts[0]),
				},
			}},
		},
	}
	s.opts["leaf"] = leafOpts
	s.leaf = startServer(t, leafOpts)

	// Wait for cluster formation and gateway connections.
	waitForCluster(t, s.clusterA[:], 10*time.Second)
	waitForCluster(t, s.clusterB[:], 10*time.Second)
	waitForGateways(t, s.clusterA[0], 1, 10*time.Second)

	// Cleanup in reverse order.
	t.Cleanup(func() {
		s.leaf.Shutdown()
		s.leaf.WaitForShutdown()
		for i := 1; i >= 0; i-- {
			s.clusterB[i].Shutdown()
			s.clusterB[i].WaitForShutdown()
		}
		for i := 1; i >= 0; i-- {
			s.clusterA[i].Shutdown()
			s.clusterA[i].WaitForShutdown()
		}
	})
}

// KillNode shuts down a named node.
func (s *SuperclusterTopology) KillNode(name string) error {
	srv := s.getServer(name)
	srv.Shutdown()
	srv.WaitForShutdown()
	return nil
}

// RestartNode restarts a previously killed node. Waits up to 10s
// for cluster rejoin.
func (s *SuperclusterTopology) RestartNode(name string) error {
	opts := s.opts[name]
	if opts == nil {
		panic("RestartNode: unknown node " + name)
	}
	// Re-use the same store dir for state recovery.
	ns, err := natsserver.NewServer(opts)
	if err != nil {
		return fmt.Errorf("restart %s: %w", name, err)
	}
	ns.Start()
	if !ns.ReadyForConnections(10 * time.Second) {
		return fmt.Errorf("restart %s: not ready after 10s", name)
	}
	s.setServer(name, ns)
	return nil
}

// DisconnectLeaf shuts down the leaf node.
func (s *SuperclusterTopology) DisconnectLeaf() error {
	s.leaf.Shutdown()
	s.leaf.WaitForShutdown()
	return nil
}

// ReconnectLeaf restarts the leaf node. Waits up to 10s for
// leaf connection to establish.
func (s *SuperclusterTopology) ReconnectLeaf() error {
	opts := s.opts["leaf"]
	ns, err := natsserver.NewServer(opts)
	if err != nil {
		return fmt.Errorf("reconnect leaf: %w", err)
	}
	ns.Start()
	if !ns.ReadyForConnections(10 * time.Second) {
		return fmt.Errorf("reconnect leaf: not ready after 10s")
	}
	s.leaf = ns
	return nil
}

func (s *SuperclusterTopology) getServer(
	name string,
) *natsserver.Server {
	switch name {
	case "a1":
		return s.clusterA[0]
	case "a2":
		return s.clusterA[1]
	case "b1":
		return s.clusterB[0]
	case "b2":
		return s.clusterB[1]
	default:
		panic("getServer: unknown node " + name)
	}
}

func (s *SuperclusterTopology) setServer(
	name string, ns *natsserver.Server,
) {
	switch name {
	case "a1":
		s.clusterA[0] = ns
	case "a2":
		s.clusterA[1] = ns
	case "b1":
		s.clusterB[0] = ns
	case "b2":
		s.clusterB[1] = ns
	default:
		panic("setServer: unknown node " + name)
	}
}

// --- Utility functions ---

func startServer(
	t *testing.T, opts *natsserver.Options,
) *natsserver.Server {
	t.Helper()
	ns, err := natsserver.NewServer(opts)
	if err != nil {
		t.Fatalf("start %s: %v", opts.ServerName, err)
	}
	ns.Start()
	if !ns.ReadyForConnections(10 * time.Second) {
		t.Fatalf("%s not ready after 10s", opts.ServerName)
	}
	return ns
}

func allocatePorts(t *testing.T, count int) []int {
	t.Helper()
	if count <= 0 {
		panic("allocatePorts: count must be positive")
	}
	ports := make([]int, count)
	for i := 0; i < count; i++ {
		l, err := net.Listen("tcp", "127.0.0.1:0")
		if err != nil {
			t.Fatalf("allocate port %d: %v", i, err)
		}
		ports[i] = l.Addr().(*net.TCPAddr).Port
		l.Close()
	}
	return ports
}

func parseURL(t *testing.T, port int) *url.URL {
	t.Helper()
	u, err := url.Parse(
		fmt.Sprintf("nats://127.0.0.1:%d", port),
	)
	if err != nil {
		t.Fatalf("parse URL port %d: %v", port, err)
	}
	return u
}

func waitForCluster(
	t *testing.T,
	servers []*natsserver.Server,
	timeout time.Duration,
) {
	t.Helper()
	if len(servers) == 0 {
		panic("waitForCluster: servers must not be empty")
	}
	if timeout <= 0 {
		panic("waitForCluster: timeout must be positive")
	}
	deadline := time.Now().Add(timeout)
	for time.Now().Before(deadline) {
		allReady := true
		for _, ns := range servers {
			if ns.NumRoutes() < len(servers)-1 {
				allReady = false
				break
			}
		}
		if allReady {
			return
		}
		time.Sleep(100 * time.Millisecond)
	}
	t.Fatal("waitForCluster: cluster did not form within timeout")
}

func waitForGateways(
	t *testing.T,
	ns *natsserver.Server,
	expectedCount int,
	timeout time.Duration,
) {
	t.Helper()
	if expectedCount <= 0 {
		panic("waitForGateways: expectedCount must be positive")
	}
	deadline := time.Now().Add(timeout)
	for time.Now().Before(deadline) {
		// NumOutboundGateways counts gateway connections, not routes.
		if ns.NumOutboundGateways() >= expectedCount {
			return
		}
		time.Sleep(100 * time.Millisecond)
	}
	t.Fatalf(
		"waitForGateways: expected %d gateways, got %d",
		expectedCount, ns.NumOutboundGateways(),
	)
}

// e2e/features/linear_test.go
// Tests a basic sequential workflow A→B through the full stack.
// Methodology: register workflow, start run, workers execute steps,
// verify completion status and event history. Runs against all
// enabled topologies via RunE2E.
package features

import (
	"testing"
	"time"

	"github.com/danmestas/dagnats/dag"
	"github.com/danmestas/dagnats/e2e/harness"
	"github.com/danmestas/dagnats/engine"
	"github.com/danmestas/dagnats/observe"
	"github.com/danmestas/dagnats/protocol"
	"github.com/danmestas/dagnats/worker"
	"github.com/nats-io/nats.go"
)

func TestLinearWorkflow(t *testing.T) {
	harness.RunE2E(t, func(t *testing.T, nc *nats.Conn) {
		tel := observe.NewNoopTelemetry()

		// Start orchestrator.
		orch := engine.NewOrchestrator(nc, tel)
		orch.Start()
		t.Cleanup(func() { orch.Stop() })

		// Register workers.
		harness.SubscribeWorker(
			t, nc, "task-a",
			func(tc worker.TaskContext) error {
				return tc.Complete([]byte(`"a-done"`))
			},
		)
		harness.SubscribeWorker(
			t, nc, "task-b",
			func(tc worker.TaskContext) error {
				return tc.Complete([]byte(`"b-done"`))
			},
		)

		// Build and run workflow.
		svc := harness.NewTestService(t, nc)
		name := harness.UniqueName(t, "linear")
		wb := dag.NewWorkflow(name)
		a := wb.Task("a", "task-a")
		wb.Task("b", "task-b").After(a)
		wfDef, err := wb.Build()
		if err != nil {
			t.Fatalf("Build: %v", err)
		}
		runID := harness.RegisterAndStart(t, svc, wfDef, nil)

		// Positive: workflow completes.
		run := harness.WaitForRunStatus(
			t, svc, runID,
			dag.RunStatusCompleted, 15*time.Second,
		)
		if run.Status != dag.RunStatusCompleted {
			t.Fatalf("expected completed, got %s", run.Status)
		}

		// Positive: both steps completed.
		if run.Steps["a"].Status != dag.StepStatusCompleted {
			t.Fatalf("step a: %s", run.Steps["a"].Status)
		}
		if run.Steps["b"].Status != dag.StepStatusCompleted {
			t.Fatalf("step b: %s", run.Steps["b"].Status)
		}

		// Positive: correct event sequence in history.
		harness.AssertHistoryContains(t, svc, runID,
			protocol.EventWorkflowStarted,
			protocol.EventStepCompleted,
			protocol.EventStepCompleted,
			protocol.EventWorkflowCompleted,
		)

		// Negative: step outputs are preserved.
		if string(run.Steps["a"].Output) != `"a-done"` {
			t.Fatalf(
				"step a output: %s", string(run.Steps["a"].Output),
			)
		}
	})
}
